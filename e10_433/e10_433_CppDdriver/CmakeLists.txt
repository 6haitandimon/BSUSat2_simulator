cmake_minimum_required(VERSION 3.13)

include(pico_sdk_import.cmake)
project(E10_DRIVER C CXX ASM)

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)


pico_sdk_init()

file(GLOB SRC_FILE ${PROJECT_SOURCE_DIR}/src/*.cpp)
file(GLOB E10_SRC_FILE ${PROJECT_SOURCE_DIR}/e10_433/*.cpp)

add_executable(E10_DRIVER
            #src/main
            ${SRC_FILE}
            #e10_433_cpp
            ${E10_SRC_FILE}
)

target_include_directories(E10_DRIVER PUBLIC e10_433/include)


target_link_libraries(E10_DRIVER
  pico_stdlib
  hardware_i2c
  # Uncomment when using Pico W
  # pico_cyw43_arch_none
)


pico_enable_stdio_usb(E10_DRIVER 1)

# enable usb output, disable uart output
pico_enable_stdio_usb(E10_DRIVER 1)
pico_enable_stdio_uart(E10_DRIVER 0)

# create map/bin/hex/uf2 file etc.
pico_add_extra_outputs(E10_DRIVER)
