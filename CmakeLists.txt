cmake_minimum_required(VERSION 3.13)

include(pico_sdk_import.cmake)
project(MatherBoard C CXX ASM)

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

pico_sdk_init()

add_subdirectory(PCA9554)
add_subdirectory(MCP2515)
add_subdirectory(INA3221)
add_subdirectory(INA219)

add_executable(MatherBoard
        main.cpp
)

pico_enable_stdio_usb(MatherBoard 1)
pico_enable_stdio_uart(MatherBoard 0)

pico_add_extra_outputs(MatherBoard)

target_link_libraries(MatherBoard
  pico_stdlib
  hardware_adc
  PCA9554
  MCP2515
  INA3221
  INA219
)
