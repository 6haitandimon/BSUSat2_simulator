cmake_minimum_required(VERSION 3.13)

include(pico_sdk_import.cmake)
project(INA3221_driver C CXX ASM)

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

pico_sdk_init()

set(HEADERS headers/ina3221.h)
set(SOURCES source/ina3221.c)

add_library(INA3221 ${SOURCES} ${HEADERS})
target_include_directories(INA3221 PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})

target_link_libraries(INA3221
  hardware_i2c
)
